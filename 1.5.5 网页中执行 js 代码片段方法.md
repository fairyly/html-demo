# 网页中执行 js 代码片段方法

就像 jsfiddle codepen 那样的 可以执行代码片段的页面

## eval(): 执行 js 代码

```
eval(code)
```



## Function()

```
var func = new Function(code)
func.call(this)
```

## demo 

```
<html>
 <head>
  <meta http-equiv="Content-Type" content="text/html;charset=UTF-8">
 </head>
 <body onload="draw();">
 <table>
 <tr>
 <td>
   <textarea rows="30" cols="60" id="code">
   //一小段代码，循环，画三个园，点击运行按钮看看右边输出效果
   for(var i=1;i<=3;i++){
      ctx.beginPath();
        ctx.arc(95,50*i,40,0,2*Math.PI);
        ctx.stroke();
   }
   </textarea>
   <BR>
   <input type="button" onclick="run();" value="运行">
 </td>
 <td>
   <canvas id="canvas" width="800" height="500">
   </canvas>
 </td>
 </tr>
 </body>
</html>
<script type="application/javascript">
    var ctx;
    function draw() {
      var canvas = document.getElementById("canvas");
      if (canvas.getContext) {
        ctx = canvas.getContext("2d");
      }
    }
    function run(){
      var  val = document.getElementById("code").value;
      // eval(val);
      var func = new Function(val)
      func.call(this)
    }
    var test = function t(x,y){
      alert(x+y);
    }
</script>

```
