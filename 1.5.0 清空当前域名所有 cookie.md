# 清空当前域名所有 cookie



```
(function clearAllCookie() {
                var keys = document.cookie.match(/[^ =;]+(?=\=)/g);
                if(keys) {
                    for(var i = keys.length; i--;)
                        document.cookie = keys[i] + '=0;expires=' + new Date(0).toUTCString()
                }
 window.location.reload()
            })()
```
