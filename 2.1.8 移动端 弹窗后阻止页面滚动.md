# 微信内打开网页 ，移动端 弹窗后阻止页面滚动


```
document.body.style.overflow = 'hidden';
function move(e){
  e.preventDefault();
  e.stopPropagation()
}
document.body.addEventListener("touchmove",move,{ passive: false, capture: true });//禁止页面滑动

# remove
function move(e){
  e.preventDefault();
  e.stopPropagation()
}
document.body.removeEventListener("touchmove",move,{ capture: true });
```
