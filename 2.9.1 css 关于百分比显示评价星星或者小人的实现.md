        function diagram(num) {
            var arr = document.getElementsByClassName("blue");
            for (let i = 0; i < arr.length; i++) {
                if (i + 1 <= num) {
                    arr[i].style.width = "58px";
                } else if (i + 1 == Math.ceil(num)) {
                    arr[i].style.width = parseInt(num * 10 - parseInt(num) * 10) / 10 * 58 + "px";
                } else {
                    arr[i].style.width = "0px";
                }
            }
